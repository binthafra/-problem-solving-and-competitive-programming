// find the largest number from a given list of integers

#include<stdio.h>
int main()
{
    int i, n, lar, elem;
    printf ("Enter total number of elements \n");
    scanf ("%d", &elem);
    printf ("Enter first number \n");
    scanf ("%d", &n);
    lar = n;
    for (i=1; i<= elem -1 ; i++)
    {
        printf ("\n Enter another number \n");
        scanf ("%d",&n);
        if (n>lar)
        lar=n;
    }
    printf ("\n The largest number is %d", lar);
    return 0;
}
